(with-open-file (*standard-output* "nul" :direction 
								   :output :if-exists :supersede) 
				  (ql:quickload :drakma)
				  (ql:quickload :com.informatimago.common-lisp.html-parser))
(defvar website "http://mislab.kuoe0.ch")
(let ((html (drakma:http-request website :external-format-in :utf-8))
	  (list-obj)
	  (ret nil))
		(setf list-obj (com.informatimago.common-lisp.html-parser.parse-html:parse-html-string html)) 
		(princ "---HTML---") 
		(terpri) 
		(princ html) 
		(terpri) 
		(princ "---LIST---") 
		(terpri) 
		(princ list-obj) 
		(terpri))
